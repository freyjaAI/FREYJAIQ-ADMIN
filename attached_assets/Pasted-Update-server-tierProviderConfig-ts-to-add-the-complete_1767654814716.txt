export function getTierProviders(tierId: string, category: ProviderCategory): ProviderConfig[] {
  const config = TIER_CONFIGURATIONS[tierId] || TIER_CONFIGURATIONS.tier_1;
  
  switch (category) {
    case 'propertyOwnership':
      return config.propertyOwnershipProviders;
    case 'contactEnrichment':
      return config.contactEnrichmentProviders;
    case 'addressValidation':
